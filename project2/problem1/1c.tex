$$
\begin{aligned}
	max(M) &= (4+8+2)e^2-2e
	\\&=14e^2-2e
\end{aligned}
$$

$$
L=max(|f'(y,t|)=2
$$

$$
\begin{aligned}
	h&=\frac{(.1)(2)(L)}{M(e^{L(t_i-a)}-1)}
	\\&=\frac{.4}{(14e^2-2e)(e^2-1)}

\end{aligned}
$$
